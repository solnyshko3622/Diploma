-- Analysis of monthly active user growth SELECT u.id, u.username, COUNT(t.id) AS transaction_count, SUM(t.amount) AS total_spent FROM users u JOIN transactions t ON u.id = t.user_id WHERE t.status = 'completed' AND t.created_at > CURRENT_DATE - INTERVAL '30 days' GROUP BY 1, 2 ORDER BY 4 DESC;
